//6.3. Configure a submissão do formulário para que, ao ser submetido um novo item ele seja  adicionado na lista. (1V)

let form = document.querySelector('form');
    form.addEventListener('submit', function(event) {
        event.preventDefault();
        let newItemText = document.querySelector('#myItem').value;
        if (newItemText.trim() !== '') {
            let newItem = document.createElement('li');
            newItem.textContent = newItemText + ' com sucesso!';
            document.querySelector('#myList').appendChild(newItem);
            document.querySelector('#myItem').value = '';
        }
    });
